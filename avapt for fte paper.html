plt.rc('xtick', labelsize='xx-large')
plt.rc('ytick', labelsize='xx-large')

fig = plt.figure(figsize=(14, 9))
ax = fig.add_subplot(1, 1, 1)
temp_df = pd.DataFrame()
temp_df['MissRate'] = avapt_df_noms['MissRate']
temp_df['LogCardinality'] = np.log2(avapt_df_noms['Cardinality'].values)
temp_df.reset_index(inplace=True)
v_bars = [5, 25, 90]
x_min = 0
x_max = 100
h_bars = [4, 6]
y_min = 0
y_max = 15
for h_bar in h_bars:
    ax.plot([x_min, x_max], [h_bar, h_bar], color='0.65', ls=':')
for v_bar in v_bars:
    ax.plot([v_bar, v_bar], [y_min, y_max], color='0.65', ls=':')
group = []
for i in range(len(temp_df)):
    f1 = 1
    f2 = 1
    if temp_df.iloc[i]['MissRate'] <= v_bars[0]:
        f1 = 1
    elif temp_df.iloc[i]['MissRate'] <= v_bars[1]:
        f1 = 2
    elif temp_df.iloc[i]['MissRate'] <= v_bars[2]:
        f1 = 3
    else:
        f1 = 4
    if temp_df.iloc[i]['LogCardinality'] <= h_bars[0]:
        f2 = 1
    elif temp_df.iloc[i]['LogCardinality'] <= h_bars[1]:
        f2 = 2
    else:
        f2 = 3
    group.append(3*(f1-1) + f2)
temp_df['group'] = group
groups = list(temp_df['group'].unique())
for g in groups:
    temp_df2 = temp_df[temp_df['group'] == g]
    ax.scatter(temp_df2['MissRate'], temp_df2['LogCardinality'],  marker='o', s=70)
ax.set_xlabel('Missing Rate in Percent', fontsize=30)
ax.set_ylabel('Log (Cardinality)', fontsize=30)
plt.tight_layout()
plt.savefig('C:\\Users\\bigebr\\Desktop\\papers (mine)\\fte\\images\\miss_rate_vs_card.pdf', dpi=1000)
